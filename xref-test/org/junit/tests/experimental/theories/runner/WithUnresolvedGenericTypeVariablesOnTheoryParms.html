164         assertThat(result, hasFailureContaining("unresolved type variable B"));
165         assertThat(result, hasFailureContaining("unresolved type variable C"));
166         assertThat(result, hasFailureContaining("unresolved type variable D"));
167         assertThat(result, hasFailureContaining("unresolved type variable E"));

169         assertThat(result, hasFailureContaining("unresolved type variable G"));
170     }
