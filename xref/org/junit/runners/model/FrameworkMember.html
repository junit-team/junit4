6   /**
7    * Parent class for {@link FrameworkField} and {@link FrameworkMethod}
8    *
9    * @since 4.7
10   */
11  public abstract class FrameworkMember<T extends FrameworkMember<T>> implements
12          Annotatable {
13      abstract boolean isShadowedBy(T otherMember);
14  
15      boolean isShadowedBy(List<T> members) {
16          for (T each : members) {
17              if (isShadowedBy(each)) {
18                  return true;
19              }
20          }
21          return false;
22      }
23  
24      protected abstract int getModifiers();
25  
26      /**
27       * Returns true if this member is static, false if not.
28       */
29      public boolean isStatic() {
30          return Modifier.isStatic(getModifiers());
31      }
32  
33      /**
34       * Returns true if this member is public, false if not.
35       */
36      public boolean isPublic() {
37          return Modifier.isPublic(getModifiers());
38      }
39  
40      public abstract String getName();
41  
42      public abstract Class<?> getType();
43  
44      public abstract Class<?> getDeclaringClass();
45  }
