6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;

10  /**
11   * Annotating an array or iterable-typed field or method with &#064;DataPoints
12   * will cause the values in the array or iterable given to be used as potential
13   * parameters for theories in that class when run with the
14   * {@link org.junit.experimental.theories.Theories Theories} runner.
15   * <p>
16   * DataPoints will only be considered as potential values for parameters for
17   * which their types are assignable. When multiple sets of DataPoints exist with
18   * overlapping types more control can be obtained by naming the DataPoints using
19   * the value of this annotation, e.g. with
20   * <code>&#064;DataPoints({"dataset1", "dataset2"})</code>, and then specifying
21   * which named set to consider as potential values for each parameter using the
22   * {@link org.junit.experimental.theories.FromDataPoints &#064;FromDataPoints}
23   * annotation.
24   * <p>
25   * Parameters with no specified source (i.e. without &#064;FromDataPoints or
26   * other {@link org.junit.experimental.theories.ParametersSuppliedBy
27   * &#064;ParameterSuppliedBy} annotations) will use all DataPoints that are
28   * assignable to the parameter type as potential values, including named sets of
29   * DataPoints.
30   * <p>
31   * DataPoints methods whose array types aren't assignable from the target
32   * parameter type (and so can't possibly return relevant values) will not be
33   * called when generating values for that parameter. Iterable-typed datapoints
34   * methods must always be called though, as this information is not available
35   * here after generic type erasure, so expensive methods returning iterable
36   * datapoints are a bad idea.
37   * 
38   * <pre>
39   * &#064;DataPoints
40   * public static String[] dataPoints = new String[] { ... };
41   * 
42   * &#064;DataPoints
43   * public static String[] generatedDataPoints() {
44   *     return new String[] { ... };
45   * }
46   * 
47   * &#064;Theory
48   * public void theoryMethod(String param) {
49   *     ...
50   * }
51   * </pre>
52   * 
53   * @see org.junit.experimental.theories.Theories
54   * @see org.junit.experimental.theories.Theory
55   * @see org.junit.experimental.theories.DataPoint
56   * @see org.junit.experimental.theories.FromDataPoints
57   */
58  @Retention(RetentionPolicy.RUNTIME)
59  @Target({ FIELD, METHOD })
60  public @interface DataPoints {
61      String[] value() default {};
62  
63      Class<? extends Throwable>[] ignoredExceptions() default {};
64  }
