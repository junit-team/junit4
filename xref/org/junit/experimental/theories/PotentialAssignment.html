5   public abstract class PotentialAssignment {
6       public static class CouldNotGenerateValueException extends Exception {
7           private static final long serialVersionUID = 1L;
8           
9           public CouldNotGenerateValueException() {
10          }
11          
12          public CouldNotGenerateValueException(Throwable e) {
13              super(e);
14          }
15      }
16  
17      public static PotentialAssignment forValue(final String name, final Object value) {
18          return new PotentialAssignment() {
19              @Override
20              public Object getValue() {
21                  return value;
22              }
23  
24              @Override
25              public String toString() {
26                  return format("[%s]", value);
27              }
28  
29              @Override
30              public String getDescription() {
31                  String valueString;
32  
33                  if (value == null) {
34                      valueString = "null";
35                  } else {
36                      try {
37                          valueString = format("\"%s\"", value);
38                      } catch (Throwable e) {
39                          valueString = format("[toString() threw %s: %s]", 
40                                               e.getClass().getSimpleName(), e.getMessage());
41                      }
42                  }
43  
44                  return format("%s <from %s>", valueString, name);
45              }
46          };
47      }
48  
49      public abstract Object getValue() throws CouldNotGenerateValueException;
50  
51      public abstract String getDescription() throws CouldNotGenerateValueException;
52  }
