3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   import org.junit.experimental.theories.internal.SpecificDataPointsSupplier;
9   
10  /**
11   * Annotating a parameter of a {@link org.junit.experimental.theories.Theory
12   * &#064;Theory} method with <code>&#064;FromDataPoints</code> will limit the
13   * datapoints considered as potential values for that parameter to just the
14   * {@link org.junit.experimental.theories.DataPoints DataPoints} with the given
15   * name. DataPoint names can be given as the value parameter of the
16   * &#064;DataPoints annotation.
17   * <p>
18   * DataPoints without names will not be considered as values for any parameters
19   * annotated with &#064;FromDataPoints.
20   * <pre>
21   * &#064;DataPoints
22   * public static String[] unnamed = new String[] { ... };
23   * 
24   * &#064;DataPoints("regexes")
25   * public static String[] regexStrings = new String[] { ... };
26   * 
27   * &#064;DataPoints({"forMatching", "alphanumeric"})
28   * public static String[] testStrings = new String[] { ... }; 
29   * 
30   * &#064;Theory
31   * public void stringTheory(String param) {
32   *     // This will be called with every value in 'regexStrings',
33   *     // 'testStrings' and 'unnamed'.
34   * }
35   * 
36   * &#064;Theory
37   * public void regexTheory(&#064;FromDataPoints("regexes") String regex,
38   *                         &#064;FromDataPoints("forMatching") String value) {
39   *     // This will be called with only the values in 'regexStrings' as 
40   *     // regex, only the values in 'testStrings' as value, and none 
41   *     // of the values in 'unnamed'.
42   * }
43   * </pre>
44   * 
45   * @see org.junit.experimental.theories.Theory
46   * @see org.junit.experimental.theories.DataPoint
47   * @see org.junit.experimental.theories.DataPoints
48   */
49  @Retention(RetentionPolicy.RUNTIME)
50  @Target(ElementType.PARAMETER)
51  @ParametersSuppliedBy(SpecificDataPointsSupplier.class)
52  public @interface FromDataPoints {
53      String value();
54  }
