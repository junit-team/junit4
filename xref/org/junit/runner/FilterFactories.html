1   package org.junit.runner;
2   
3   import org.junit.internal.Classes;
4   import org.junit.runner.FilterFactory.FilterNotCreatedException;
5   import org.junit.runner.manipulation.Filter;
6   
7   /**
8    * Utility class whose methods create a {@link FilterFactory}.
9    */
10  class FilterFactories {
11      /**
12       * Creates a {@link Filter}.
13       *
14       * A filter specification is of the form "package.of.FilterFactory=args-to-filter-factory" or
15       * "package.of.FilterFactory".
16       *
17       * @param request the request that will be filtered
18       * @param filterSpec the filter specification
19       * @throws org.junit.runner.FilterFactory.FilterNotCreatedException
20       */
21      public static Filter createFilterFromFilterSpec(Request request, String filterSpec)
22              throws FilterFactory.FilterNotCreatedException {
23          Description topLevelDescription = request.getRunner().getDescription();
24          String[] tuple;
25  
26          if (filterSpec.contains("=")) {
27              tuple = filterSpec.split("=", 2);
28          } else {
29              tuple = new String[]{ filterSpec, "" };
30          }
31  
32          return createFilter(tuple[0], new FilterFactoryParams(topLevelDescription, tuple[1]));
33      }
34  
35      /**
36       * Creates a {@link Filter}.
37       *
38       * @param filterFactoryFqcn The fully qualified class name of the {@link FilterFactory}
39       * @param params The arguments to the {@link FilterFactory}
40       */
41      public static Filter createFilter(String filterFactoryFqcn, FilterFactoryParams params)
42              throws FilterFactory.FilterNotCreatedException {
43          FilterFactory filterFactory = createFilterFactory(filterFactoryFqcn);
44  
45          return filterFactory.createFilter(params);
46      }
47  
48      /**
49       * Creates a {@link Filter}.
50       *
51       * @param filterFactoryClass The class of the {@link FilterFactory}
52       * @param params             The arguments to the {@link FilterFactory}
53       *
54       */
55      public static Filter createFilter(Class<? extends FilterFactory> filterFactoryClass, FilterFactoryParams params)
56              throws FilterFactory.FilterNotCreatedException {
57          FilterFactory filterFactory = createFilterFactory(filterFactoryClass);
58  
59          return filterFactory.createFilter(params);
60      }
61  
62      static FilterFactory createFilterFactory(String filterFactoryFqcn) throws FilterNotCreatedException {
63          Class<? extends FilterFactory> filterFactoryClass;
64  
65          try {
66              filterFactoryClass = Classes.getClass(filterFactoryFqcn).asSubclass(FilterFactory.class);
67          } catch (Exception e) {
68              throw new FilterNotCreatedException(e);
69          }
70  
71          return createFilterFactory(filterFactoryClass);
72      }
73  
74      static FilterFactory createFilterFactory(Class<? extends FilterFactory> filterFactoryClass)
75              throws FilterNotCreatedException {
76          try {
77              return filterFactoryClass.getConstructor().newInstance();
78          } catch (Exception e) {
79              throw new FilterNotCreatedException(e);
80          }
81      }
82  }
