001package org.junit.runners.parameterized;
002
003import org.junit.runner.Runner;
004import org.junit.runners.model.InitializationError;
005
006/**
007 * A {@link ParametersRunnerFactory} that creates
008 * {@link BlockJUnit4ClassRunnerWithParameters}.
009 * 
010 * @since 4.12
011 */
012public class BlockJUnit4ClassRunnerWithParametersFactory implements
013        ParametersRunnerFactory {
014    public Runner createRunnerForTestWithParameters(TestWithParameters test)
015            throws InitializationError {
016        return new BlockJUnit4ClassRunnerWithParameters(test);
017    }
018}
